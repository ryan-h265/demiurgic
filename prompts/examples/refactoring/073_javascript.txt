function searchArray(arr, target) {
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] === target) {
            return i;
        }
    }
    for (let i = 0; i < arr.length; i++) {
        if (arr[i].id === target) {
            return i;
        }
    }
    return -1;
}
