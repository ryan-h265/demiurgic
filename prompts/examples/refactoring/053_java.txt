public List<Integer> filterPositive(List<Integer> numbers) {
    List<Integer> result = new ArrayList<>();
    for (int i = 0; i < numbers.size(); i++) {
        if (numbers.get(i) > 0) {
            result.add(numbers.get(i));
        }
    }
    return result;
}
