```javascript
function quickSort(arr, lo = 0, hi = arr.length - 1) {
  if (lo >= hi) return arr;
  const p = partition(arr, lo, hi);
  quickSort(arr, lo, p - 1);
  quickSort(arr, p + 1, hi);
  return arr;
}

function partition(arr, lo, hi) {
  const pivot = arr[hi];
  let i = lo;
  for (let j = lo; j < hi; j++) {
    if (arr[j] <= pivot) {
      [arr[i], arr[j]] = [arr[j], arr[i]];
      i++;
    }
  }
  [arr[i], arr[hi]] = [arr[hi], arr[i]];
  return i;
}
```
**Explanation:** In-place Lomuto partition quicksort. Recursively sorts subarrays around a pivot; partition ensures elements ≤ pivot are left of it.
**Complexity:** Time: O(n log n) average, O(n²) worst; Space: O(log n) recursion.
**Example:**
Input: [3,1,4,1]
Output: [1,1,3,4]
**Notes:**
- Randomize pivot to mitigate worst case.
- Not stable; equal elements may reorder.
- Tail recursion not guaranteed; iterative conversion for huge arrays.
