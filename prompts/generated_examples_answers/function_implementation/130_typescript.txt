```typescript
export function evalPostfix(tokens: string[]): number {
  if (!tokens.length) throw new Error('empty expression');
  const stack: number[] = [];
  for (const t of tokens) {
    if (['+', '-', '*', '/'].includes(t)) {
      const b = stack.pop();
      const a = stack.pop();
      if (a === undefined || b === undefined) throw new Error('invalid expression');
      switch (t) {
        case '+': stack.push(a + b); break;
        case '-': stack.push(a - b); break;
        case '*': stack.push(a * b); break;
        case '/': stack.push((a / b) | 0); break; // truncate toward 0
      }
    } else {
      stack.push(Number(t));
    }
  }
  if (stack.length !== 1) throw new Error('invalid expression');
  return stack[0];
}
```
**Explanation:** Processes tokens left-to-right, using a stack for operands; operators pop the top two operands, apply the operation, and push the result. Valid postfix expressions end with exactly one value on the stack.
**Complexity:** Time: O(n); Space: O(n) for the stack.
**Example:**
Input: ['2','1','+','3','*']
Output: 9
**Notes:**
- Division uses integer truncation via bitwise-or; replace with `a / b` for floating results.
- Validate tokens to avoid `NaN` inputs.
- Extend with additional operators by adding cases.
