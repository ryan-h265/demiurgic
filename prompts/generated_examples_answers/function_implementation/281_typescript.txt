```typescript
export function gcd(a: number, b: number): number {
  a = Math.abs(a);
  b = Math.abs(b);
  while (b !== 0) {
    const tmp = a % b;
    a = b;
    b = tmp;
  }
  return a;
}
```
**Explanation:** Euclidean algorithm iteratively replaces (a, b) with (b, a % b) until b becomes zero; the remaining a is the GCD. Takes absolute values to handle negatives.
**Complexity:** Time: O(log min(a, b)); Space: O(1).
**Example:**
Input: gcd(54, 24)
Output: 6
**Notes:**
- If both inputs are zero, returns zero; define behavior if you need an error instead.
- For arrays of numbers, fold this function across the collection.
- Use bigint if you need exact results for very large integers.
