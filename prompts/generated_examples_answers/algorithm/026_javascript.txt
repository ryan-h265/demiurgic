```javascript
// In-place quicksort (Lomuto partition) with comments
function quickSort(arr, lo = 0, hi = arr.length - 1) {
  if (lo >= hi) return arr;
  const p = partition(arr, lo, hi);
  quickSort(arr, lo, p - 1);
  quickSort(arr, p + 1, hi);
  return arr;
}

function partition(arr, lo, hi) {
  const pivot = arr[hi];
  let i = lo;
  for (let j = lo; j < hi; j++) {
    if (arr[j] <= pivot) {
      [arr[i], arr[j]] = [arr[j], arr[i]];
      i++;
    }
  }
  [arr[i], arr[hi]] = [arr[hi], arr[i]];
  return i;
}
```
**Explanation:** Recursively sorts subarrays around a pivot. `partition` places elements ≤ pivot to the left and returns the pivot’s final index, then quicksort recurses on the left/right partitions.
**Complexity:** Time: O(n log n) average, O(n²) worst when partitions are unbalanced; Space: O(log n) recursion stack.
**Example:**
Input: [3, 1, 4, 1]
Output: [1, 1, 3, 4]
**Notes:**
- Randomize the pivot (swap with a random index) to mitigate worst cases.
- Tail-call optimize or switch to insertion sort on tiny partitions for speed.
- Not stable; equal elements may reorder.
